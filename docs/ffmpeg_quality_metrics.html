251        """
252        Get the version of ffmpeg
253        """
254        cmd = ["ffmpeg", "-version"]
255        stdout, _ = run_command(cmd)

257        # ffmpeg version 7.1.1 Copyright (c) 2000-2025 the FFmpeg developers
258        # ...

261        # ffmpeg version n7.1-latest-linux64-gpl-7.1
262        version_str = stdout.split("\n")[0].split(" ")[2]
263
264        # Clean the version string by removing the "-static" suffix if it exists
265        version_str = version_str.split("-")[0]
266
267        # Clean a "n" at the beginning of the version string if it exists
268        version_str = version_str.lstrip("n")
269
270        try:
271            return parse_version(version_str)
272        except Exception:
273            logger.warning(f"Could not parse ffmpeg version: {version_str}")

277    def get_framerate(input_file: str) -> float:
278        """Parse the FPS from the input file.
279
280        Args:
281            input_file (str): Input file path
282
283        Raises:
284            FfmpegQualityMetricsError: A generic error
285
286        Returns:
287            float: The FPS parsed
288        """
289        cmd = ["ffmpeg", "-nostdin", "-y", "-i", input_file]
290
291        output = run_command(cmd, allow_error=True)
292        pattern = re.compile(r"(\d+(\.\d+)?) fps")
293        try:
294            if pattern_ret := pattern.search(str(output)):
295                match = pattern_ret.groups()[0]
296                return float(match)
297        except Exception:
298            pass
299
300        raise FfmpegQualityMetricsError(f"could not parse FPS from file {input_file}!")

336    def calculate(
337        self,
338        metrics: List[MetricName] = ["ssim", "psnr"],
339        vmaf_options: Union[VmafOptions, None] = None,
340    ) -> Dict[MetricName, SingleMetricData]:
341        """Calculate one or more metrics.
342
343        Args:
344            metrics (list, optional): A list of metrics to calculate.
345                Possible values are ["ssim", "psnr", "vmaf"].
346                Defaults to ["ssim", "psnr"].
347            vmaf_options (dict, optional): VMAF-specific options. Uses defaults if not specified.
348
349        Raises:
350            FfmpegQualityMetricsError: In case of an error
351            e: A generic error
352
353        Returns:
354            dict: A dictionary of per-frame info, with the key being the metric name and the value being a dict of frame numbers ('n') and metric values.
355        """
356        if not metrics:
357            raise FfmpegQualityMetricsError("No metrics specified!")
358
359        # check available metrics
360        for metric_name in metrics:
361            filter_name = self.METRIC_TO_FILTER_MAP.get(metric_name, None)
362            if filter_name not in self.POSSIBLE_FILTERS:
363                raise FfmpegQualityMetricsError(f"No such metric '{metric_name}'")
364            if filter_name not in self.available_filters:
365                raise FfmpegQualityMetricsError(
366                    f"Your ffmpeg version does not have the filter '{filter_name}'"
367                )
368
369        # set VMAF options specifically
370        if "vmaf" in metrics:
371            self._check_libvmaf_availability()
372            self.vmaf_options = self.DEFAULT_VMAF_OPTIONS
373            # override with user-supplied options
374            if vmaf_options:
375                for key, value in vmaf_options.items():
376                    if value is not None:
377                        self.vmaf_options[key] = value  # type: ignore
378            self._set_vmaf_model_path(self.vmaf_options["model_path"])
379
380        ffmpeg_version = FfmpegQualityMetrics.get_ffmpeg_version()
381        if ffmpeg_version is None or ffmpeg_version < parse_version("7.1"):
382            logger.warning(
383                "FFmpeg version is less than 7.1. Using deprecated scale2ref filter. Please update to 7.1 or higher."
384            )
385            filter_chains = [
386                f"[1][0]scale2ref=flags={self.scaling_algorithm}[dist][ref]",
387                "[dist]settb=AVTB,setpts=PTS-STARTPTS[distpts]",
388                "[ref]settb=AVTB,setpts=PTS-STARTPTS[refpts]",
389            ]
390        else:
391            # ffmpeg 7.1 or higher: scale2ref filter is deprecated
392            # input 0: ref, input 1: dist --> swapped for scale filter
393            filter_chains = [
394                f"[1][0]scale=rw:rh:flags={self.scaling_algorithm}[dist]",
395                "[dist]settb=AVTB,setpts=PTS-STARTPTS[distpts]",
396                "[0]settb=AVTB,setpts=PTS-STARTPTS[refpts]",
397            ]
398
399        # generate split filters depending on the number of models
400        n_splits = len(metrics)
401        if n_splits > 1:
402            for source in ["dist", "ref"]:
403                suffixes = "".join([f"[{source}{n}]" for n in range(1, n_splits + 1)])
404                filter_chains.extend(
405                    [
406                        f"[{source}pts]split={n_splits}{suffixes}",
407                    ]
408                )
409
410        # special case, only one metric:
411        if n_splits == 1:
412            metric_name = metrics[0]
413            filter_chains.extend(
414                [
415                    f"[distpts][refpts]{self._get_filter_opts(self.METRIC_TO_FILTER_MAP[metric_name])}"
416                ]
417            )
418        # all other cases:
419        else:
420            for n, metric_name in zip(range(1, n_splits + 1), metrics):
421                filter_chains.extend(
422                    [
423                        f"[dist{n}][ref{n}]{self._get_filter_opts(self.METRIC_TO_FILTER_MAP[metric_name])}"
424                    ]
425                )
426
427        try:
428            output = self._run_ffmpeg_command(filter_chains, desc=", ".join(metrics))
429            self._read_temp_files(metrics)
430            if output:
431                self._read_ffmpeg_output(output, metrics)
432            else:
433                raise FfmpegQualityMetricsError("ffmpeg output is empty!")
434        except Exception as e:
435            raise e
436        finally:
437            self._cleanup_temp_files()
438
439        # return only those data entries containing values
440        return cast(
441            Dict[MetricName, SingleMetricData],
442            {k: v for k, v in self.data.items() if v},
443        )

445    def _get_libvmaf_filter_opts(self) -> str:
446        """
447        Returns:
448
449            str: A string to use for VMAF in ffmpeg filter chain
450        """
451        # we only have one model, and its path parameter is not optional
452        all_model_params: Dict[str, str] = {
453            "path": win_vmaf_model_path_check(self.vmaf_model_path)
454        }
455
456        # add further model parameters
457        for model_param in self.vmaf_options["model_params"]:
458            key, value = model_param.split("=")
459            all_model_params[key] = value
460
461        all_model_params_str = "\\:".join(
462            f"{k}={v}" for k, v in all_model_params.items()
463        )
464
465        vmaf_opts: Dict[str, str] = {
466            "model": all_model_params_str,
467            "log_path": win_path_check(self.temp_files["libvmaf"]),
468            "log_fmt": "json",
469            "n_threads": str(self.vmaf_options["n_threads"]),
470            "n_subsample": str(self.vmaf_options["n_subsample"]),
471        }
472
473        if self.vmaf_options["features"]:
474            features = []
475            for feature in self.vmaf_options["features"]:
476                if not feature.startswith("name"):
477                    feature = f"name={feature}"
478                features.append(feature.replace(":", "\\:"))
479            vmaf_opts["feature"] = "|".join(features)
480
481        vmaf_opts_string = ":".join(
482            f"{k}={v}" for k, v in vmaf_opts.items() if v is not None
483        )
484
485        return vmaf_opts_string

542    def _parse_ffmpeg_metadata_output(
543        self, ffmpeg_output: str, metric_name: Literal["vif", "msad"]
544    ) -> None:
545        """
546        Parse the filter output written to ffmpeg's metadata output
547
548        Args:
549            ffmpeg_output (str): The output of ffmpeg's stderr
550            metric_name (Literal["vif", "msad"]): The name of the metric to parse
551        """
552        # Example for VIF:
553        #
554        # [Parsed_metadata_4 @ 0x7f995cd08640] frame:1    pts:1       pts_time:0.0401x
555        # [Parsed_metadata_4 @ 0x7f995cd08640] lavfi.vif.scale.0=0.263582
556        # [Parsed_metadata_4 @ 0x7f995cd08640] lavfi.vif.scale.1=0.560129
557        # [Parsed_metadata_4 @ 0x7f995cd08640] lavfi.vif.scale.2=0.626596
558        # [Parsed_metadata_4 @ 0x7f995cd08640] lavfi.vif.scale.3=0.682183
559        #
560        # Example for MSAD:
561        #
562        # [Parsed_metadata_6 @ 0x10ad04ea0] lavfi.msad.msad.Y=0.029998
563        # [Parsed_metadata_6 @ 0x10ad04ea0] lavfi.msad.msad.U=0.019501
564        # [Parsed_metadata_6 @ 0x10ad04ea0] lavfi.msad.msad.V=0.026455
565        # [Parsed_metadata_6 @ 0x10ad04ea0] lavfi.msad.msad_avg=0.025318
566
567        lines = [line.strip() for line in ffmpeg_output.split("\n")]
568        current_frame = None
569        frame_data: Dict[str, float] = {}
570
571        for line in lines:
572            if not line.startswith("[Parsed_metadata"):
573                continue
574
575            fields = line.split(" ")
576
577            # a new frame appears
578            if fields[3].startswith("frame"):
579                # if we have data already
580                if frame_data:
581                    self.data[metric_name].append(frame_data)
582
583                # get the frame number and reset the frame data
584                current_frame = int(fields[3].split(":")[1])
585                frame_data = {"n": current_frame}
586                continue
587
588            # no frame was set, or no VIF info present
589            if current_frame is None or not fields[3].startswith(
590                f"lavfi.{metric_name}"
591            ):
592                continue
593
594            # we have a frame
595            key, value = fields[3].split("=")
596            key = key.replace(f"lavfi.{metric_name}.", "").replace(".", "_").lower()
597            frame_data[key] = round(float(value), 3)
598
599        # append final frame data
600        if frame_data:
601            self.data[metric_name].append(frame_data)

616    def _run_ffmpeg_command(
617        self, filter_chains: List[str] = [], desc: str = ""
618    ) -> Union[str, None]:
619        """
620        Run the ffmpeg command to get the quality metrics.
621        The filter chains must be specified manually.
622        'desc' can be a human readable description for the progress bar.
623
624        Returns:
625            Union[str, None]: The output of ffmpeg's stderr
626        """
627        if not self.framerate:
628            ref_framerate, dist_framerate = self._get_framerates()
629        else:
630            ref_framerate = self.framerate
631            dist_framerate = self.framerate
632
633        cmd = [
634            "ffmpeg",
635            "-nostdin",
636            "-nostats",
637            "-y",
638            "-threads",
639            str(self.threads),
640            "-r",
641            str(ref_framerate),
642            "-i",
643            self.ref,
644            "-itsoffset",
645            str(self.dist_delay),
646            "-r",
647            str(dist_framerate),
648            "-i",
649            self.dist,
650            "-filter_complex",
651            ";".join(filter_chains),
652            "-an",
653            "-f",
654            "null",
655            NUL,
656        ]
657
658        if self.progress:
659            logger.debug(quoted_cmd(cmd))

336    def calculate(
337        self,
338        metrics: List[MetricName] = ["ssim", "psnr"],
339        vmaf_options: Union[VmafOptions, None] = None,
340    ) -> Dict[MetricName, SingleMetricData]:
341        """Calculate one or more metrics.
342
343        Args:
344            metrics (list, optional): A list of metrics to calculate.
345                Possible values are ["ssim", "psnr", "vmaf"].
346                Defaults to ["ssim", "psnr"].
347            vmaf_options (dict, optional): VMAF-specific options. Uses defaults if not specified.
348
349        Raises:
350            FfmpegQualityMetricsError: In case of an error
351            e: A generic error
352
353        Returns:
354            dict: A dictionary of per-frame info, with the key being the metric name and the value being a dict of frame numbers ('n') and metric values.
355        """
356        if not metrics:
357            raise FfmpegQualityMetricsError("No metrics specified!")
358
359        # check available metrics
360        for metric_name in metrics:
361            filter_name = self.METRIC_TO_FILTER_MAP.get(metric_name, None)
362            if filter_name not in self.POSSIBLE_FILTERS:
363                raise FfmpegQualityMetricsError(f"No such metric '{metric_name}'")
364            if filter_name not in self.available_filters:
365                raise FfmpegQualityMetricsError(
366                    f"Your ffmpeg version does not have the filter '{filter_name}'"
367                )
368
369        # set VMAF options specifically
370        if "vmaf" in metrics:
371            self._check_libvmaf_availability()
372            self.vmaf_options = self.DEFAULT_VMAF_OPTIONS
373            # override with user-supplied options
374            if vmaf_options:
375                for key, value in vmaf_options.items():
376                    if value is not None:
377                        self.vmaf_options[key] = value  # type: ignore
378            self._set_vmaf_model_path(self.vmaf_options["model_path"])
379
380        ffmpeg_version = FfmpegQualityMetrics.get_ffmpeg_version()
381        if ffmpeg_version is None or ffmpeg_version < parse_version("7.1"):
382            logger.warning(
383                "FFmpeg version is less than 7.1. Using deprecated scale2ref filter. Please update to 7.1 or higher."
384            )
385            filter_chains = [
386                f"[1][0]scale2ref=flags={self.scaling_algorithm}[dist][ref]",
387                "[dist]settb=AVTB,setpts=PTS-STARTPTS[distpts]",
388                "[ref]settb=AVTB,setpts=PTS-STARTPTS[refpts]",
389            ]
390        else:
391            # ffmpeg 7.1 or higher: scale2ref filter is deprecated
392            # input 0: ref, input 1: dist --> swapped for scale filter
393            filter_chains = [
394                f"[1][0]scale=rw:rh:flags={self.scaling_algorithm}[dist]",
395                "[dist]settb=AVTB,setpts=PTS-STARTPTS[distpts]",
396                "[0]settb=AVTB,setpts=PTS-STARTPTS[refpts]",
397            ]
398
399        # generate split filters depending on the number of models
400        n_splits = len(metrics)
401        if n_splits > 1:
402            for source in ["dist", "ref"]:
403                suffixes = "".join([f"[{source}{n}]" for n in range(1, n_splits + 1)])
404                filter_chains.extend(
405                    [
406                        f"[{source}pts]split={n_splits}{suffixes}",
407                    ]
408                )
409
410        # special case, only one metric:
411        if n_splits == 1:
412            metric_name = metrics[0]
413            filter_chains.extend(
414                [
415                    f"[distpts][refpts]{self._get_filter_opts(self.METRIC_TO_FILTER_MAP[metric_name])}"
416                ]
417            )
418        # all other cases:
419        else:
420            for n, metric_name in zip(range(1, n_splits + 1), metrics):
421                filter_chains.extend(
422                    [
423                        f"[dist{n}][ref{n}]{self._get_filter_opts(self.METRIC_TO_FILTER_MAP[metric_name])}"
424                    ]
425                )
426
427        try:
428            output = self._run_ffmpeg_command(filter_chains, desc=", ".join(metrics))

432            else:
433                raise FfmpegQualityMetricsError("ffmpeg output is empty!")
434        except Exception as e:
435            raise e
436        finally:
437            self._cleanup_temp_files()
438
439        # return only those data entries containing values
440        return cast(
441            Dict[MetricName, SingleMetricData],
442            {k: v for k, v in self.data.items() if v},
443        )

784        Returns:
785            dict: A dictionary with stats, each key being a metric name and each value being a dictionary with the stats for every submetric. The stats are: 'average', 'median', 'stdev', 'min', 'max'.

803                }
